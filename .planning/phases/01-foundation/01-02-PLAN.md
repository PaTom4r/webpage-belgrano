```typescript
import type { Metadata } from 'next'
import { Geist, Inter } from 'next/font/google'
import { Analytics } from '@vercel/analytics/react'
import './globals.css'

const geistSans = Geist({
  subsets: ['latin'],
  variable: '--font-geist',
})

const inter = Inter({
  subsets: ['latin'],
  variable: '--font-inter',
  display: 'swap',
})

export const metadata: Metadata = {
  metadataBase: new URL('https://belgrano.cl'),
  title: {
    template: '%s | Belgrano',
    default: 'Belgrano — IA, Marketing & Estrategia',
  },
  description: 'Agencia chilena de IA aplicada, marketing digital y estrategia de negocio.',
  openGraph: {
    siteName: 'Belgrano',
    locale: 'es_CL',
    type: 'website',
  },
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="es" className={`${geistSans.variable} ${inter.variable}`}>
      {/*
        GSAP hydration fix: ScrollTrigger adds borderTopStyle: solid to body during
        client render. Pre-applying it here makes SSR and CSR outputs identical,
        preventing React hydration errors.
        See: https://gsap.com/community/forums/topic/43281-hydration-error-in-nextjs-15/
      */}
      <body style={{ borderTopStyle: 'solid' }}>
        {children}
        <Analytics />
      </body>
    </html>
  )
}
```

Key points:
- `Geist({ variable: '--font-geist' })` creates the CSS variable that globals.css `--font-sans` chains to
- Both `geistSans.variable` and `inter.variable` applied as classes on `<html>` — not `body`
- `metadataBase: new URL('https://belgrano.cl')` — all relative OG paths become absolute
- `<Analytics />` from `@vercel/analytics/react` (zero-config on Vercel)
- `<body style={{ borderTopStyle: 'solid' }}>` — GSAP hydration fix (MUST stay here, never remove)

**`src/components/layout/section.tsx`**:
```typescript
import { Container } from './container'

// Section wrapper with light and dark variants.
// dark=true: bg-dark text-bg (dark section for contrast, e.g. Hero, CTA)
// dark=false: bg-bg text-text (default white section)
// bg-bg-section: #f9fafb alternate for visual separation without going full dark
export function Section({
  children,
  className = '',
  dark = false,
  id,
}: {
  children: React.ReactNode
  className?: string
  dark?: boolean
  id?: string
}) {
  return (
    <section
      id={id}
      className={`py-16 sm:py-20 lg:py-24 ${
        dark ? 'bg-dark text-bg' : 'bg-bg text-text'
      } ${className}`}
    >
      <Container>{children}</Container>
    </section>
  )
}
```

Note: `bg-dark`, `text-bg`, `bg-bg`, `text-text` are Tailwind utilities generated from the @theme tokens in globals.css. They will resolve correctly as long as Plan 01's globals.css is in place.
